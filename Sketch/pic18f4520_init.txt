#include <stdio.h>
#include <stdlib.h>
#include <htc.h>
#include "init.h"

void Init(void);
void InitCPU(void);
void InitTimers(void);
void InitUART(void);


void Init(void)
{
InitCPU();
InitTimers();
InitUART();
}

void InitCPU(void)
{
  INTCON = 0;         // Interruprs Disabled
  INTCON2 = 0x84;     // TMR0 = High priority
                        // All PORTB pull-ups are disabled
  INTCON3 = 0;
  RCON = 0x80;
  PIE1 = PIE2 = 0;
  PIR1 = PIR2 = 0;
  ADCON0 = 0;
  ADCON1 = 0xF;      // Digital I/O
  ADCON2 = 0;
  CCP1CON = 0;       
  SSPCON1 = 0;        // SSP disabled
  SSPSTAT = 0;
  CMCON = 0x07;
// Initialization of ports
